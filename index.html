// Azure8 用 Google Apps Script // スプレッドシートID: 15XO-aqOmiQHTaXp81tExFoD6pAZXCNFF65bnfW5l6Gg const SPREADSHEET_ID = '15XO-aqOmiQHTaXp81tExFoD6pAZXCNFF65bnfW5l6Gg'; const ALLOWED_EMAIL = 'your-email@gmail.com'; // ← あなたのGmailアドレスに変更 function doPost(e) { try { // セキュリティチェック const userEmail = Session.getActiveUser().getEmail(); if (userEmail !== ALLOWED_EMAIL) { return ContentService.createTextOutput( JSON.stringify({ error: 'Unauthorized' }) ).setMimeType(ContentService.MimeType.JSON); } const requestData = JSON.parse(e.postData.contents); const action = requestData.action || 'add'; if (action === 'add') { return addCard(requestData.data); } else if (action === 'update') { return updateCard(requestData.data); } else if (action === 'delete') { return deleteCard(requestData.data); } else if (action === 'getData') { return getAllCards(); } return ContentService.createTextOutput( JSON.stringify({ success: true }) ).setMimeType(ContentService.MimeType.JSON); } catch (error) { Logger.log('Error: ' + error.toString()); return ContentService.createTextOutput( JSON.stringify({ error: error.toString() }) ).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const userEmail = Session.getActiveUser().getEmail(); if (userEmail !== ALLOWED_EMAIL) { return ContentService.createTextOutput( JSON.stringify({ error: 'Unauthorized' }) ).setMimeType(ContentService.MimeType.JSON); } return getAllCards(); } catch (error) { return ContentService.createTextOutput( JSON.stringify({ error: error.toString() }) ).setMimeType(ContentService.MimeType.JSON); } } function addCard(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet; // カテゴリー別にシートを作成 const sheetName = data.category; sheet = ss.getSheetByName(sheetName); if (!sheet) { // シートが存在しない場合は作成 sheet = ss.insertSheet(sheetName); // ヘッダー行を追加 sheet.appendRow([ '登録日', 'カテゴリー', '番組名', '媒体名', '名前', '会社名・役職', '電話番号', 'メールアドレス', 'メモ', 'タイムスタンプ' ]); // ヘッダー行のスタイル設定 const headerRange = sheet.getRange(1, 1, 1, 10); headerRange.setBackground('#2d5a8c'); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); headerRange.setHorizontalAlignment('center'); } // データを追加 sheet.appendRow([ data.date || '', data.category || '', data.program || '', data.media || '', data.name || '', data.company || '', data.phone || '', data.email || '', data.memo || '', new Date().toISOString() ]); // 列幅を自動調整 for (let i = 1; i <= 10; i++) { sheet.autoResizeColumn(i); } return ContentService.createTextOutput( JSON.stringify({ success: true, message: 'データを保存しました' }) ).setMimeType(ContentService.MimeType.JSON); } function updateCard(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(data.category); if (!sheet) { return ContentService.createTextOutput( JSON.stringify({ error: 'Sheet not found' }) ).setMimeType(ContentService.MimeType.JSON); } const dataRange = sheet.getDataRange(); const values = dataRange.getValues(); // 名前で検索して更新 for (let i = 1; i < values.length; i++) { if (values[i][4] === data.name) { // 5列目（名前） sheet.getRange(i + 1, 1).setValue(data.date || ''); sheet.getRange(i + 1, 2).setValue(data.category || ''); sheet.getRange(i + 1, 3).setValue(data.program || ''); sheet.getRange(i + 1, 4).setValue(data.media || ''); sheet.getRange(i + 1, 5).setValue(data.name || ''); sheet.getRange(i + 1, 6).setValue(data.company || ''); sheet.getRange(i + 1, 7).setValue(data.phone || ''); sheet.getRange(i + 1, 8).setValue(data.email || ''); sheet.getRange(i + 1, 9).setValue(data.memo || ''); sheet.getRange(i + 1, 10).setValue(new Date().toISOString()); break; } } return ContentService.createTextOutput( JSON.stringify({ success: true, message: '更新しました' }) ).setMimeType(ContentService.MimeType.JSON); } function deleteCard(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(data.category); if (!sheet) { return ContentService.createTextOutput( JSON.stringify({ error: 'Sheet not found' }) ).setMimeType(ContentService.MimeType.JSON); } const dataRange = sheet.getDataRange(); const values = dataRange.getValues(); // 名前と登録日で検索して削除 for (let i = 1; i < values.length; i++) { if (values[i][4] === data.name && values[i][0] === data.date) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput( JSON.stringify({ success: true, message: '削除しました' }) ).setMimeType(ContentService.MimeType.JSON); } function getAllCards() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheets = ss.getSheets(); const allCards = []; sheets.forEach(sheet => { const sheetName = sheet.getName(); // システムシートをスキップ if (sheetName.startsWith('Sheet')) return; const data = sheet.getDataRange().getValues(); // ヘッダー行をスキップ for (let i = 1; i < data.length; i++) { if (data[i][4]) { // 名前が入力されている行のみ allCards.push({ date: data[i][0], category: data[i][1], program: data[i][2], media: data[i][3], name: data[i][4], company: data[i][5], phone: data[i][6], email: data[i][7], memo: data[i][8], timestamp: data[i][9] }); } } }); return ContentService.createTextOutput( JSON.stringify(allCards) ).setMimeType(ContentService.MimeType.JSON); } // テスト用関数 function testAddCard() { const testData = { date: '2025-04-07', category: 'TV', program: 'ニュース23', media: 'TBS', name: 'テスト太郎', company: 'テスト株式会社', phone: '03-1234-5678', email: 'test@example.com', memo: 'テストメモ' }; addCard(testData); Logger.log('テストデータを追加しました'); }
